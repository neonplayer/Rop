actor mastergun : Weapon
{
  Weapon.AmmoUse 0
  +WEAPON.NOALERT
  +WEAPON.NOAUTOFIRE
  +WEAPON.AMMO_OPTIONAL
  +WEAPON.ALT_AMMO_OPTIONAL
  +WEAPON.CHEATNOTWEAPON
  +INVENTORY.UNDROPPABLE
  +WEAPON.NOAUTOFIRE
  States
  {
  Ready:
    TNT1 A 1 A_WeaponReady(wrf_allowreload)
    Loop
  Deselect:
    TNT1 A 0 A_Lower
    Loop
  Select:
    TNT1 A 0 A_Raise
    Loop
  Fire:
    TNT1 A 0 ACS_ExecuteAlways(mop_possess)
	goto ready
  AltFire:
    TNT1 A 0 ACS_ExecuteAlways(mop_findmarines)
	goto ready
  Spawn:
    TNT1 A 1 Thing_Remove (0)
    Stop
  reload:
    TNT1 A 0 a_jumpifinventory("floorhugtoken", 1, "reload2")
    TNT1 A 0 a_giveinventory("floorhugtoken", 1)
    TNT1 A 35 a_print("Floor hugging enabled.", 3.0)
    TNT1 A 0 a_changeflag(floorhugger, 1)
	goto ready
  reload2:
    TNT1 A 0 a_takeinventory("floorhugtoken", 1)
    TNT1 A 35 a_print("Floor hugging disabled.", 3.0)
    TNT1 A 0 a_changeflag(floorhugger, 0)
	goto ready
  }
}

actor floorhugtoken : inventory { inventory.maxamount 1 }
