actor possessedROPlostsoul : possessedmonster
{
  player.morphweapon "ROPsoulcharger"
  Health 75
  radius 14
  height 40
  Mass 50
  Speed 0.35
  PainChance 256
  +FLOAT
  +NOGRAVITY
  +MISSILEMORE
  +DONTFALL
  +NOICEDEATH
  +noblooddecals
  AttackSound "skull/melee"
  PainSound "skull/pain"
  DeathSound "skull/death"
  ActiveSound "skull/active"
  RenderStyle SoulTrans
  Obituary "%o was rammed by %k."
  scale 0.85
  Player.SoundClass Soul
  player.viewheight 16
  bloodtype "soulblood"
  tag "Lost Soul"
  states
  {
  Spawn:
    SKUL AB 10 Bright
    Loop
  See:
    SKUL AB 6 Bright
    goto spawn
  Missile:
    SKUL C 1 Bright
	SKUL D 2 Bright
    Goto Spawn
  melee:
    SKUL A 0 a_jumpifinventory("soulflametoken", 15, "melee2")
	goto spawn
  melee2:
    SKUL CD 1 bright
	SKUL A 0 a_jumpifinventory("soulflametoken", 1, "melee2")
	goto spawn
  Pain:
    SKUL E 3 Bright
    SKUL E 3 Bright A_Pain
    Goto spawn
  }
}
